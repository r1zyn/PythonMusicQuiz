FUNCTION  get_marks(tries<-0, type<-"", correct_places<-0):
    IF type = "multichoice" or type = "word"
        THEN
        IF tries = 1
            THEN
            return 10
        ENDIF
        elif tries = 2:  
            return 5
        elif tries = 3: 
            return 3
        elif tries = 4:  
            return 1
        ELSE
            return 0
    ENDIF
    elif type = "true/false": 
        IF tries = 1
            THEN
            return 10
        ENDIF
        elif tries = 2: 
            return 5
        ELSE
            return 0
    elif type = "order": 
        IF correct_places = 4
            THEN
            return 10
        ENDIF
        elif correct_places = 3:  
            return 5
        elif correct_places = 2:  
            return 3
        elif correct_places = 1: 
            return 1
        ELSE 
            return 0
ENDFUNCTION

FUNCTION  run_questions(topic):
    tries<-0
    marks<-0
ENDFUNCTION

    IF topic = "dynamics"

        OUTPUT "a Getting louder\nb Getting quiter\nc Getting faster\nd Getting slower"
INPUT answer<-input("Your answer: ")
        tries += 1
    ENDIF

        WHILE answer != "a" DO
